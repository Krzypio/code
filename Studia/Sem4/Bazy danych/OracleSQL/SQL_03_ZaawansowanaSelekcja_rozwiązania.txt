ZAAWANSOWANA SELEKCJA DANYCH - ZADANIA
Zad 1.
SELECT nazwisko, SUBSTR(etat, 1, 2) || ID_PRAC AS kod FROM pracownicy

Zad 2.
SELECT nazwisko, TRANSLATE(nazwisko, 'KLM', 'XXX') AS wojna_literom FROM pracownicy

Zad 3.
SELECT nazwisko FROM pracownicy WHERE SUBSTR(nazwisko, 1, LENGTH(nazwisko)/2) LIKE ('%L%')

Zad 4.
SELECT nazwisko, ROUND(placa_pod*1.15,0) AS podwyzka FROM pracownicy

Zad 5.
SELECT nazwisko, placa_pod, 
placa_pod*0.2 AS inwestycja, 
placa_pod*0.2 * POWER(1+10/100, 10) AS kapital, 
placa_pod*0.2 * POWER(1+10/100, 10) - placa_pod*0.2 AS zysk 
FROM pracownicy

Zad 6.
SELECT nazwisko, zatrudniony, FLOOR(MONTHS_BETWEEN(DATE '2000-01-01', zatrudniony)/12) AS staz_w_2000 FROM pracownicy

Zad 7.
SELECT nazwisko, TO_CHAR(zatrudniony, 'MONTH, MM YYYY') AS data_zatrudnienia FROM pracownicy

Zad 8.
SELECT TO_CHAR(SYSDATE, 'DAY') AS dzis FROM dual

Zad 9.
SELECT nazwa, adres, 
CASE
  WHEN adres LIKE 'MIELZYNSKIEGO%' THEN 'STARE MIASTO'
  WHEN adres LIKE 'STRZELECKA%' THEN 'STARE MIASTO'
  WHEN adres LIKE 'PIOTROWO%' THEN 'NOWE MIASTO' 
  WHEN adres LIKE 'WLODKOWICA%' THEN 'GRUNWALD'  
END AS dzielnica 
FROM ZESPOLY

Zad 10.
SELECT nazwisko, placa_pod, 
CASE 
  WHEN placa_pod>480 THEN 'Powyżej 480'
  WHEN placa_pod=480 THEN 'Dokładnie 480'
  WHEN placa_pod<480 THEN 'Poniżej 480' 
END AS prog 
FROM pracownicy

Zad 11.
SELECT nazwisko, placa_pod,
DECODE(SIGN(placa_pod-480),
  	1, 'Ponad 480',
	0, 'Dokladnie 480',
  'Ponizej 480'
) AS prog
FROM pracownicy
